```C
// 선형검색 (보초법 ver) (search 함수의 while을 for문으로 바꾸기)
#include <stdio.h>
#include <stdlib.h>

// 요소의 개수가 n인 배열 a에서 key와 일치하는 요소를 선형 검색(보초법)
int search(int a[], int n, int key)
{
	int i;
	a[n] = key; // 보초를 추가
	for(i=0; i < n ;i++)
	{

		if (a[i] == key)
			break; // 원하는 키 값을 찾은 경우
	}
	return i == n ? -1 : i; // 검색 성공,실패 판단
}

int main(void)
{
	int i, nx, ky, idx;
	int* x;
	puts("선형 검색");
	printf("요소 개수 : ");
	scanf("%d", &nx);
	x = (int*)malloc((nx + 1) * sizeof(int));

	for (i = 0; i < nx; i++)
	{
		printf("x[%d] : ", i);
		scanf("%d", &x[i]);
	}
	printf("검색값 : ");
	scanf("%d", &ky);
	idx = search(x, nx, ky);

	if (idx == -1)
		puts("검색에 실패했습니다.");
	else
		printf("%d(은)는 x[%d]에 있습니다.\n", ky, idx);
	free(x);

	return 0;
}
```
